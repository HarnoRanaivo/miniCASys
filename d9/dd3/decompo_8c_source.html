14 LUM* decomposition(Matrix *m)
15 {
16  int n = nbLignes(m);
17  E somme=0.;
18  E somme2=0.;
19  LUM* raiponce = MALLOCN(raiponce, 2);
20 
21  if (raiponce != NULL)
22  {
23  Matrix * l = newMatrix(n,n); // L
24  Matrix * u = newMatrix(n,n); // U
25 
26  setElt(u,1,1,getElt(m,1,1));
27 
28 
29  for (int j = 2 ; j<=n ; j++)
30  {
31  setElt(u,1,j,getElt(m,1,j));
32  setElt(l,j,1,getElt(m,j,1)/getElt(m,1,1));
33  }
34 
35  for (int i = 2 ; i<n ; i++)
36  {
37  for (int k = 1 ; k<i ; k++)
38  {
39  somme += getElt(l,i,k)*getElt(u,k,i);
40  }
41 
42  setElt(u,i,i,getElt(m,i,i)-somme);
43  somme=0.;
44 
45  for (int j = i+1 ; j<=n ; j++)
46  {
47  for (int k = 1 ; k<i ; k++)
48  {
49  somme += getElt(l,i,k)*getElt(u,k,j);
50  somme2 += getElt(l,j,k)*getElt(u,k,i);
51  }
52 
53  setElt(u,i,j,getElt(m,i,j)-somme);
54  setElt(l,j,i,1./getElt(u,i,i)*(getElt(m,j,i)-somme2));
55  }
56 
57  somme=0.;
58  somme2=0.;
59  }
60 
61  for (int k = 1 ; k<n ; k++)
62  {
63  somme += getElt(l,n,k)*getElt(u,k,n);
64  }
65 
66  setElt(u,n,n,getElt(m,n,n)-somme);
67 
68  for (int i = 1; i <= n; i++)
69  setElt(l,i,i,1.);
70 
71  raiponce[0]= l;
72  raiponce[1] = u;
73  }
74  else
75  perror("malloc");
76 
77  return raiponce;
78 }
