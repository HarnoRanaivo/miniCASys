299 Matrix * traiterCommande(Commande c, char * arguments, const Variables * v)
300 {
301  Matrix * m = NULL;
302  char buffer1[64] = { '\0' };
303  char buffer2[64] = { '\0' };
304  E buffer3;
305  int buffer4;
306 
307  switch(c)
308  {
309  case CM_NEW :
310  if (contientMatriceValide(arguments))
311  m = recupererMatrice(arguments, v);
312  else
313  fprintf(stderr, "%s : matrice non valide.\n", arguments);
314  break;
315 
316  case CM_ADD :
317 
318  case CM_SUB :
319 
320  case CM_MULM :
321 
322  case CM_SOL :
323  /* Partie commune aux quatres commandes CM_ADD, CM_SUB, CM_MULM,
324  * CM_SOL.
325  */
326  if (sscanf(arguments, " %63[^, ]%*[,]%63s", buffer1, buffer2) == 2)
327  {
328  const Donnee * d1 = obtenirDonnee(v, buffer1);
329  const Donnee * d2 = obtenirDonnee(v, buffer2);
330  if (d1 == NULL || d2 == NULL)
331  {
332  if (d1 == NULL)
333  fprintf(stderr, "%s n'existe pas.\n", buffer1);
334  if (d2 == NULL)
335  fprintf(stderr, "%s n'existe pas.\n", buffer2);
336  }
337  else if (!estMatrice(d1) || !estMatrice(d2))
338  {
339  if (!estMatrice(d1))
340  fprintf(stderr, "%s n'est pas une matrice.\n", buffer1);
341  if (!estMatrice(d2))
342  fprintf(stderr, "%s, n'est pas une matrice.\n", buffer2);
343  }
344  else
345  {
346  const Matrix * m1 = matriceDonnee(d1);
347  const Matrix * m2 = matriceDonnee(d2);
348 
349  /* Addition et soustraction. */
350  if (c == CM_ADD || c == CM_SUB)
351  {
352  if (nbLignes(m1) == nbLignes(m2) && nbColonnes(m1) == nbColonnes(m2))
353  m = c == CM_ADD ? addition(m1, m2) : soustraction(m1, m2);
354  else
355  fprintf(stderr, "Les matrices n'ont pas la même taille.\n");
356  }
357  /* Multiplication. */
358  else if (c == CM_MULM)
359  {
360  if (nbColonnes(m1) == nbLignes(m2))
361  m = multiplication(m1, m2);
362  else
363  fprintf(stderr, "Les matrices n'ont pas les bonnes tailles.\n");
364  }
365  /* Résolution. */
366  else
367  {
368  if (nbLignes(m1) == nbLignes(m2) && nbColonnes(m2) == 1)
369  {
370  Matrix * m0 = copieMatrice(m1);
371  Matrix * b0 = copieMatrice(m2);
372  Matrix * solution = newMatrix(nbLignes(m1), 1);
373 
374  gauss(m0, b0, solution);
375  deleteMatrix(m0);
376  deleteMatrix(b0);
377 
378  m = solution;
379  }
380  else
381  fprintf(stderr, "Les matrices n'ont pas les bonnes tailles.\n");
382  }
383  }
384  }
385  break;
386 
387  case CM_MULS :
388  /* L'utilisateur a donné une variable et un flottant. */
389  if (sscanf(arguments, " %63[^, ]%*[,]%f", buffer1, &buffer3) == 2)
390  {
391  const Donnee * d1 = obtenirDonnee(v, buffer1);
392  if (d1 == NULL)
393  fprintf(stderr, "%s n'existe pas.\n", buffer1);
394  else if (!estMatrice(d1))
395  fprintf(stderr, "%s n'est pas une matrice.\n", buffer1);
396  else
397  m = multiplierScalaire(matriceDonnee(d1), buffer3);
398  }
399  /* L'utilisateur a donné deux variables. */
400  else if (sscanf(arguments, " %63[^, ]%*[,]%31s", buffer1, buffer2) == 2)
401  {
402  const Donnee * d1 = obtenirDonnee(v, buffer1);
403  const Donnee * d2 = obtenirDonnee(v, buffer2);
404  if (d1 == NULL || d2 == NULL)
405  {
406  if (d1 == NULL)
407  fprintf(stderr, "%s n'existe pas.\n", buffer1);
408  if (d2 == NULL)
409  fprintf(stderr, "%s n'existe pas.\n", buffer2);
410  }
411  else if (!estMatrice(d1) || !estE(d2))
412  {
413  if (!estMatrice(d1))
414  fprintf(stderr, "%s n'est pas une matrice.\n", buffer1);
415  if (!estE(d2))
416  fprintf(stderr, "%s n'est pas un flottant.\n", buffer2);
417  }
418  else
419  m = multiplierScalaire(matriceDonnee(d1), eDonnee(d2));
420  }
421  else
422  fprintf(stderr, "Erreur.\n");
423  break;
424 
425  case CM_EXP :
426  if (sscanf(arguments, " %63[^, ]%*[,]%d", buffer1, &buffer4) == 2)
427  {
428  const Donnee * d1 = obtenirDonnee(v, buffer1);
429 
430  if (d1 == NULL)
431  fprintf(stderr, "%s n'existe pas.\n", buffer1);
432  else if (!estMatrice(d1))
433  fprintf(stderr, "%s n'est pas une matrice.\n", buffer1);
434  else if (buffer4 < 0)
435  fprintf(stderr, "L'exposant doit être positif.\n");
436  else if (nbLignes(matriceDonnee(d1)) != nbColonnes(matriceDonnee(d1)))
437  fprintf(stderr, "La matrice n'est pas carrée.\n");
438  else
439  m = exponentiation(matriceDonnee(d1), buffer4);
440  }
441  else if (sscanf(arguments, " %63[^, ]%*[,]%31s", buffer1, buffer2) == 2)
442  {
443  const Donnee * d1 = obtenirDonnee(v, buffer1);
444  const Donnee * d2 = obtenirDonnee(v, buffer2);
445 
446  if (d1 == NULL || d2 == NULL)
447  {
448  if (d1 == NULL)
449  fprintf(stderr, "%s n'existe pas.\n", buffer1);
450  if (d2 == NULL)
451  fprintf(stderr, "%s n'existe pas.\n", buffer2);
452  }
453  else if (!estMatrice(d1) || !estE(d2))
454  {
455  if (!estMatrice(d1))
456  fprintf(stderr, "%s n'est pas une matrice.\n", buffer1);
457  if (!estE(d2))
458  fprintf(stderr, "%s n'est pas une matrice.\n", buffer2);
459  }
460  else if (eDonnee(d2) < 0)
461  fprintf(stderr, "L'exposant doit être positif.\n");
462  else if (nbLignes(matriceDonnee(d1)) != nbColonnes(matriceDonnee(d1)))
463  fprintf(stderr, "La matrice n'est pas carrée.\n");
464  else
465  m = exponentiation(matriceDonnee(d1), eDonnee(d2));
466  }
467  break;
468 
469  case CM_TSP :
470 
471  case CM_INV :
472  if (sscanf(arguments, " %63[^, ]", buffer1) == 1)
473  {
474  const Donnee * d1 = obtenirDonnee(v, buffer1);
475 
476  if (d1 == NULL)
477  fprintf(stderr, "%s n'existe pas.\n", buffer1);
478  else if (!estMatrice(d1))
479  fprintf(stderr, "%s n'est pas une matrice.\n", buffer1);
480  else
481  m = c == CM_TSP ? transpose(matriceDonnee(d1)) : inverseM(matriceDonnee(d1));
482  }
483  break;
484 
485  default :
486  break;
487  }
488 
489  return m;
490 }

492 static Variables * lancerDecomposition(Variables * v, char * copie, char * morceaux[2])
493 {
494  Commande c = CM_INCONNU;
495  Donnee *d;
496  supprimerEspaces(copie);
497  c = rechercherCommande(copie);
498 
499  if (c == CM_LU)
500  {
501  char variable[32] = { '\0' };
502  if (sscanf(morceaux[0], "%31s", variable) == 1
503  && (d = obtenirDonnee(v, variable)) != NULL
504  && estMatrice(d)
505  )
506  {
507  LUM * lu = decomposition(matriceDonnee(d));
508  v = ajouterMatrice(v, "L", copieMatrice(lu[0]));
509  v = ajouterMatrice(v, "U", copieMatrice(lu[1]));
510  afficheLU(lu);
511  libererLU(lu);
512  }
513  }
514  else if (c != CM_INCONNU)
515  fprintf(stderr, "%s : Mauvaise utilisation.\n", copie);
516  else
517  fprintf(stderr, "%s : Commande inconnue.\n", copie);
518 
519  return v;
520 }

522 static Bool lancerSpeedtest(const char * buffer, const char * commande)
523 {
524  Bool continuer = VRAI;
525  char fonction[32] = { '\0' };
526  int min, max, pas, sec;
527 
528  if (sscanf(buffer, "%*s %31s %d %d %d %d", fonction, &min, &max, &pas, &sec) == 5)
529  {
530  struct timeval tv1, tv2;
531  unsigned long long temps;
532 
533  Commande c = rechercherCommande(fonction);
534  if ((c == CM_ADD || c == CM_SUB || c == CM_MULM)
535  && min > 0 && min <= max && pas > 0
536  )
537  {
538  gettimeofday(&tv1, NULL);
539  speedtest(c, min, max, pas);
540  gettimeofday(&tv2, NULL);
541 
542  temps = (tv2.tv_sec - tv1.tv_sec);
543  if (temps > sec)
544  continuer = FAUX;
545  }
546  }
547  else if (sscanf(buffer, "%*s %31s %d %d %d", fonction, &min, &max, &pas) == 4)
548  {
549  Commande c = rechercherCommande(fonction);
550  if ((c == CM_ADD || c == CM_SUB || c == CM_MULM)
551  && min > 0 && min <= max && pas > 0
552  )
553  speedtest(c, min, max, pas);
554  }
555  else
556  fprintf(stderr, "%s : Arguments invalides.\n", commande);
557 
558  return continuer;
559 }

561 static Variables * ligneDeuxParties(Variables * v, char * parties[4], Commande c)
562 {
563  supprimerEspaces(parties[0]);
564  c = rechercherCommande(parties[0]);
565 
566  if (c != CM_INCONNU || rechercherMot(parties[0], (const char * []) { "L", "U", NULL, }))
567  fprintf(stderr, "%s : Mot-clé réservé.\n", parties[0]);
568  else
569  {
570  char variable_1[32];
571  E valeur;
572  Donnee * d;
573 
574  /* Ligne de la forme "variable : flottant" ? */
575  if (sscanf(parties[1], "%f\n", &valeur) == 1)
576  {
577  char variable[32];
578 
579  if (sscanf(parties[0], "%31s", variable) == 1)
580  {
581  v = ajouterE(v, variable, valeur);
582  printf("%s :\n\t%f\n", variable, valeur);
583  }
584  }
585  /* Ligne de la forme "variable_2 : variable_1" ? */
586  else if (sscanf(parties[1], "%31s", variable_1) == 1
587  && (d = obtenirDonnee(v, variable_1)) != NULL
588  )
589  {
590  char variable[32];

833

839  break;
